-------------------------
-- Drop tables if they already exist
------------------------- 

DROP TABLE IF EXISTS Product2;
DROP TABLE IF EXISTS Company2;



---------------------------
-- Create the tables
---------------------------

create table Company2 ( 
	cid int PRIMARY KEY, 
	cname 	varchar(20),  
	city	varchar(20)); 

create table Product2 ( 
	pname 	varchar(20), 
	price	float, 
	cid	int, 
PRIMARY KEY (PName, cid), 
FOREIGN KEY (cid) REFERENCES Company2(cid) );
